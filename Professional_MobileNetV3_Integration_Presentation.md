def quantize(x, bit_width=16, frac_bits=8):
    scale = 2 ** frac_bits
    min_val = -2**(bit_width-1)
    max_val = 2**(bit_width-1) - 1
    x_q = np.round(x * scale)
    x_q = np.clip(x_q, min_val, max_val).astype(np.int16)
    return x_q

def export_weights(model, output_dir):
